  <li>Dólar americano</li>
  <li>Dólar canadense</li>
  <li>Dólar australiano</li>
  <li>Libra esterlina</li>
  <li>Yen</li>
  <li>Yuan renmimbi</li>
  <li>Franco suiço</li>
